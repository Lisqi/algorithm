// 随机一个数字 var bomb_num = parseInt(Math.random() * 100) + 1; // 范围提示 var min = 1; var max = 100; // 让用户输入 while (true) { // 弹出输入框 var n = Number(prompt(`请输入一个数字，范围${min}到${max}`)); // 判断用户输入的数字是否在范围内 if (n >= min && n <= max) { if (n > bomb_num) { alert('猜大了'); // 改变max max = n - 1; } else if (n < bomb_num) { alert('猜小了');                 // 改变min min = n + 1; } else if (n == bomb_num) { alert('爆炸！！！'); break; } } else { alert('你输入的数字不符合要求'); } }
